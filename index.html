With parameters

Parameters for customize the selector

var example3 = document.getElementById("example3");
    new dateSelect(example3, {
    separator: "/",
    months: [
        "Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio",
        "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"],
    selectsAttrs: { "class" : "selector" },
    yElementAttrs: { "id": "left" },
    mElementAttrs: { "id": "middle" },
    dElementAttrs: { "id": "rigth" }
});

Date Range

A range of dates, with a function for valid input.

— 

var example4_beg = document.getElementById("example4-begin"),
    example4_end = document.getElementById("example4-end"),
    parameters = {
    minY: 2000,
    maxY: (new Date).getFullYear(),
    selectsAttrs: { "class" : "selector" },
    yElementAttrs: { "id": "left" },
    mElementAttrs: { "id": "middle" },
    dElementAttrs: { "id": "rigth" },
    afterChange: function(e) {
        var date1 = parseInt(dateBeg.date.replace(/\-/g, ""), 10),
            date2 = parseInt(dateEnd.date.replace(/\-/g, ""), 10);
        if (date1 > date2) {
            alert("Initial date can't be greater than final date\nPlease do not do that :(");
        }
    }
};
dateBeg = new dateSelect(example4_beg, parameters);
dateEnd = new dateSelect(example4_end, parameters);
